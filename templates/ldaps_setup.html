# Script d'installation certificat LDAPS
# A executer en PowerShell ADMINISTRATEUR sur le serveur AD ({{ server }})

# Domaine: {{ domain }}
$Domain = "{{ domain }}"
$DCName = "$env:COMPUTERNAME.$Domain"
$IP = (Get-NetIPAddress -AddressFamily IPv4 | Where-Object { $_.InterfaceAlias -notlike '*Loopback*' } | Select-Object -First 1).IPAddress

Write-Host "Creation certificat pour: $Domain, $DCName, $IP" -ForegroundColor Yellow

$cert = New-SelfSignedCertificate `
    -DnsName @($Domain, $DCName, $IP, $env:COMPUTERNAME, "localhost") `
    -CertStoreLocation "Cert:\LocalMachine\My" `
    -KeyExportPolicy Exportable `
    -Provider "Microsoft RSA SChannel Cryptographic Provider" `
    -NotAfter (Get-Date).AddYears(5) `
    -KeyLength 2048 `
    -HashAlgorithm SHA256

Write-Host "SUCCESS: Certificat cree - $($cert.Thumbprint)" -ForegroundColor Green

Restart-Service NTDS -Force
Write-Host "Service AD redemarre - LDAPS actif!" -ForegroundColor Green

📋 Instructions

  1. Option automatique: Cliquez sur "Installer le certificat LDAPS" ci-dessus
  2. Option manuelle: Copiez le script et exécutez-le sur le serveur AD en PowerShell Admin
  3. Après installation, déconnectez-vous et reconnectez avec SSL
